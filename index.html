// Cấu hình URL của Google Apps Script const GAS_URL = "https://script.google.com/macros/s/AKfycbzCDGNc74gWMOqJGN74YFo92e9QEq-FBUQVkm-UAGhJeM8MMZMeetDNe8MXeYkA09tJ/exec"; let isLogin = true; // --- PHẦN 1: XỬ LÝ ĐĂNG NHẬP / ĐĂNG KÝ (Ở index.html) --- function toggleForm() { isLogin = !isLogin; const formTitle = document.getElementById('form-title'); const toggleText = document.getElementById('toggle-text'); if (formTitle) formTitle.innerText = isLogin ? "Đăng Nhập" : "Đăng Ký"; if (toggleText) toggleText.innerText = isLogin ? "Chưa có tài khoản? Đăng ký ngay" : "Đã có tài khoản? Đăng nhập"; } async function handleAuth() { const user = document.getElementById('username').value; const pass = document.getElementById('password').value; const action = isLogin ? "login" : "signup"; if (!user || !pass) return alert("Vui lòng nhập đủ thông tin!"); const btn = document.getElementById('main-btn'); const originalText = btn.innerText; btn.innerText = "Đang xử lý..."; btn.disabled = true; try { await fetch(GAS_URL, { method: "POST", mode: "no-cors", body: JSON.stringify({ action, user, pass }) }); alert(isLogin ? "Đang kiểm tra đăng nhập..." : "Đang tạo tài khoản..."); setTimeout(() => { if (isLogin) { // Lưu tên user vào localStorage để dùng cho các trang sau localStorage.setItem("username", user); window.location.href = "home.html"; } else { alert("Đăng ký hoàn tất! Giờ hãy đăng nhập nhé."); toggleForm(); btn.innerText = originalText; btn.disabled = false; } }, 2000); } catch (error) { alert("Lỗi kết nối rồi ông ơi!"); btn.innerText = originalText; btn.disabled = false; } } // --- PHẦN 2: XỬ LÝ TRANG CHỦ & ĐIỂM (Ở home.html) --- window.onload = () => { const savedUser = localStorage.getItem("username"); // Nếu đang ở trang home.html if (window.location.pathname.includes("home.html")) { if (!savedUser) { // Nếu chưa đăng nhập mà đòi vào home thì đá về index window.location.href = "index.html"; } else { // Hiển thị tên người dùng const userDisplay = document.getElementById("user-display"); const welcomeMsg = document.querySelector("h1"); if (userDisplay) userDisplay.innerText = "Chào, " + savedUser; if (welcomeMsg) welcomeMsg.innerText = `Chào mừng ${savedUser} đã quay trở lại!`; } } } async function addPost() { const user = localStorage.getItem("username"); const content = document.getElementById("post-content").value; const btn = document.getElementById("post-btn"); if (!content) return alert("Ông chưa viết nội dung mà!"); btn.innerText = "Đang lưu bài..."; btn.disabled = true; try { await fetch(GAS_URL, { method: "POST", mode: "no-cors", body: JSON.stringify({ action: "addPoint", user: user, points: 10 }) }); alert("Đăng bài thành công! +10 điểm."); document.getElementById("post-content").value = ""; // Cập nhật điểm tạm thời trên màn hình const pointElement = document.getElementById("user-points"); if (pointElement) { let currentPoints = parseInt(pointElement.innerText); pointElement.innerText = currentPoints + 10; } } catch (error) { alert("Không cộng được điểm rồi!"); } finally { btn.innerText = "Đăng bài (+10 điểm)"; btn.disabled = false; } } function logout() { localStorage.removeItem("username"); window.location.href = "index.html"; }
